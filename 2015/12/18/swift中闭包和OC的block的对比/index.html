-(void)laodData:(void (^)(NSString *html)) finished {

    dispatch_async(dispatch_get_global_queue(0, 0), ^{

    NSLog(@"耗时操作");

    dispatch_async(dispatch_get_main_queue(), ^{

        NSLog(@"准备主线陈的回调");

        finished(@"<html>");

    });

  });

}
